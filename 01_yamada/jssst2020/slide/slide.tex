 \end{itemize}

\begin{frame}\frametitle{根付き全域森(Spanning Rooted Forest)}

 \begin{block}{根付き全域森~~[川原$\cdot$湊 '12]}
  グラフ$G=(V,E)$と，
  \alert{根}と呼ばれる$V$上のノードが与えられたとき，
  $G$上の根付き全域森とは，以下の条件を満たす$G$の部分グラフ$G'=(V,E'),\ E' \subseteq E$である．
  \begin{enumerate}
   \item $G'$はサイクルを持たない． (\alert{非閉路制約})
   \item $G'$の各連結成分は，ちょうど1つの根を含む． (\alert{根付き連結制約})
  \end{enumerate}
 \end{block}

\begin{frame}\frametitle{ASPの基本構文}
\begin{itemize}
 \item ASPのプログラムは，以下の形式の\structure{ルール}の有限集合である．
\begin{block}{}
 \centering
 $a_0 ~\text{:-}~ a_1, \dots, a_m, ~\text{\tt not}~ a_{m+1}, \ldots, \text{\tt not}~ a_n.$
\end{block}
 ここで，$0 \leq m \leq n$ であり，各$a_i$はアトム，{\tt not}は\structure{デフォルトの否定}，
 “$,$”は連言を表す.“:-”の左側を\structure{ヘッド}，
	   右側を\structure{ボディ}と呼ぶ．“.”はルールの終わりを表す終端記号である．
	   \vskip 1ex
	   直感的な意味は，「\redunderline{$a_1,\ldots,a_m$がすべて成り立ち，$a_{m+1},\ldots,a_n$の}\\
	   \redunderline{それぞれが成り立たないならば，$a_0$が成り立つ}」である．

	   \vskip 2ex

 \item ヘッドが空のルールを\structure{一貫性制約}と呼ぶ．
\begin{block}{}
 \centering
 $ \text{:-} ~a_1, ~{\tt not}~ a_2.$
\end{block}
	上のルールは，「$a_1$が成り立つならば，$a_2$も成り立つ」ことを意味する．
\end{itemize}
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
%% ASPのシンタックスの説明 その2
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}\frametitle{ASPの拡張構文}
\begin{itemize}
 \item ASP言語には，組合せ問題を解くための便利な拡張構文が用意されている．
\end{itemize}
 \begin{block}{選択子}
  \[
  \{ a_1 ; \ldots ; a_n \}
  \]
  アトム集合$\{a_1,\ldots,a_n\}$の任意の部分集合が成り立つことを意味する．
 \end{block}

 \begin{block}{個数制約}
  \[
  s~ \{ a_1 ; \ldots ; a_n \} ~k
  \]
  $a_1,\ldots,a_n$のうち，$s$個以上$k$個以下が成り立つことを意味する．\\
  また，各アトムに重みを設定し，その重み和について上下限を制限する
  \structure{重み付き個数制約}({\tt \#sum})も使用できる．
 \end{block}
\end{frame}

\begin{frame}\frametitle{根付き全域森の改良符号化}
 \begin{exampleblock}{}
 \texttt{ \{inForest(X,Y)\} :- edge(X,Y).} {\Large\}}\alert{解の候補}\\
 \vskip 1em
 $
 \left.
 \begin{tabular}{l}
  \texttt{reached(R,R) :- root(R).} \\
  \texttt{reached(X,R) :- reached(Y,R), inForest(Y,X).} \\
  \texttt{reached(X,R) :- reached(Y,R), inForest(X,Y).}
 \end{tabular}
 \right\}
 $ \alert{到達可能性}\\
 \vskip 1em
 $
 \left.
 \begin{tabular}{l}
  \texttt{:- root(R), } \\
  \texttt{~~~not 1 \#sum\{ 1,X:reached(X,R) ;}\\
  \texttt{~~~~~~~~~~~~~~-1,X,Y:inForest(X,Y),}\\
  \texttt{~~~~~~~~~~~~~~~~~~~~~reached(X,R), reached(Y,R) }\\
  \texttt{~~~~~~~~~~~~~\} 1.}\\
 \end{tabular}
 \right\} 
 \begin{array}{c}
  \alert{\text{非閉路}}\\
  \alert{\text{制約}} 
 \end{array}
 $ \mbox{}\\
 \vskip 1em
 ~~\texttt{:- node(X), not 1 \{reached(X,R)\} 1.} {\Large\}}\alert{根付き連結制約}
 \end{exampleblock}
 \begin{itemize}
  \item 全部で\structure{6つ}のルールで記述可能．
 \end{itemize}
\end{frame}

\begin{frame}{解の遷移問題への拡張}

\begin{itemize}
 \item ある初期配電網構成(\structure{スタート状態})から，
	   目的配電網構成(\structure{ゴール状態})へのスイッチの
	   切替手順を求める問題へ拡張．
	   \begin{itemize}
		\item 各ステップ$t$で切替可能なスイッチの個数を$d$個以下に制限．
		\item 最短ステップ長での切替手順を求めることが目的．
	   \end{itemize}
	   \vskip 1ex
 \item 配電網における障害時の復旧予測への応用が期待される．
	   \vskip 2ex
 \item トポロジ制約のみの場合，\alert{根付き全域森遷移問題}に帰着できる．
	   \begin{itemize}
		\item スタート状態からゴール状態まで，根付き全域森の制約を満たす．
		\item \structure{遷移制約}も満たしながら移る必要がある．
	   \end{itemize}
\begin{block}{遷移制約}
 各ステップ$t$で変更可能な辺の数は$d$個以下でなければならない．
\end{block}
\end{itemize} 
\end{frame}

\begin{frame}\frametitle{根付き全域森遷移問題の解法}
 \vskip -1ex
 \begin{exampleblock}{シングルショット符号化}
  \begin{itemize}
   \item 改良符号化の自然な拡張．
   \begin{itemize}
   \item 各アトムにステップを表す項を引数に追加．
   \item スタート状態，ゴール状態，遷移制約に関するルールを追加．
   \end{itemize}
   %\item 解が存在するまでステップ長$t$を増やしながら，繰り返し解く．
   \item 各問題中の制約の大部分は共通であるため，同一の探索空間を何度も調べることになり，
		 \alert{求解効率が低下する}という問題点がある．
  \end{itemize}
 \end{exampleblock}

 \begin{alertblock}{マルチショット符号化}
  \begin{itemize}
   \item ASPシステム\textit{clingo}のマルチショットASP解法を用いて実装．
   \item ASPシステムが同様の探索失敗を避けるために獲得した学習節を(部分的に)保持することができる．
   \item 学習節によって，\alert{無駄な探索を行うことなく}，
		 制約を追加した論理プログラムを連続的に解くことができる．
  \end{itemize}
 \end{alertblock}
\begin{itemize}
 \item マルチショットASP解法は，解の遷移問題に対して，求解性能の向上が期待できる．
\end{itemize}
\end{frame}

\begin{frame}{補足 : スマートグリッド}
 \begin{itemize}
  \item \structure{スマートグリッド}とは，電力の供給側，需要側において双方向の

  \item 従来と違い，通信技術の発達により，使用状況などを
		リアルタイムに把握することが可能となった．
  \item その時に応じた最適な配電網を構成し，制御するといったことが考えられている．
		\begin{itemize}
		 \item 電力需要の変化による，配電ロスの少ない構成．
		 \item 自然エネルギーによる発電量の変動を補う構成．
		\end{itemize}
  \item ASP言語の表現力や拡張性が，こうした条件の追加に活用できる可能性がある．
 \end{itemize}
\end{frame}

\begin{frame}{補足 : 電気制約}
 \begin{itemize}
  \item \alert{電気制約}は，送電する電流$\cdot$電圧の適正範囲を保証する制約．
  \begin{itemize}
   \item 供給経路の各区間で許容電流を超えない．
   \item 電気抵抗による電圧降下が許容範囲を超えない．
   \item etc.
  \end{itemize}
  \item 電流と電圧が影響し合う\structure{実数ドメイン上の制約}によって表される．
		% \begin{itemize}
		%  		 \item 送電システム上の条件など．
		% \end{itemize}
  \item 実数ドメイン上の制約は，純粋なASPのみで扱うのは\alert{困難}．
		\begin{itemize}
		 \item 緩和問題として，変電所から供給できる家庭の数に上限をつける．
		 \item ASPMT技術により，ASPで得られた解について，
			   背景理論ソルバーと連携して実数ドメイン上の制約を調べる．
		\end{itemize}
 \end{itemize}
\end{frame}
